-- LocalScript bên trong Frame hoặc ScreenGui

local userInputService = game:GetService("UserInputService")
local frame = script.Parent  -- Tham chiếu đến Frame hoặc ScreenGui

local isFrameVisible = true  -- Biến trạng thái của Frame
local isShiftPressed = false  -- Biến kiểm tra phím Shift có được nhấn không

-- Hàm để ẩn/hiện Frame
local function toggleFrameVisibility()
	isFrameVisible = not isFrameVisible
	frame.Visible = isFrameVisible
end

-- Lắng nghe sự kiện khi một phím được nhấn
userInputService.InputBegan:Connect(function(input, gameProcessed)
	if not gameProcessed then  -- Kiểm tra xem sự kiện có bị xử lý bởi hệ thống không
		-- Kiểm tra nếu phím Shift được nhấn
		if input.KeyCode == Enum.KeyCode.LeftShift or input.KeyCode == Enum.KeyCode.RightShift then
			isShiftPressed = true
		end

		-- Kiểm tra nếu phím S được nhấn và Shift đang được giữ
		if input.KeyCode == Enum.KeyCode.S and isShiftPressed then
			toggleFrameVisibility()
		end
	end
end)

-- Lắng nghe sự kiện khi phím được thả
userInputService.InputEnded:Connect(function(input)
	-- Kiểm tra nếu Shift được thả
	if input.KeyCode == Enum.KeyCode.LeftShift or input.KeyCode == Enum.KeyCode.RightShift then
		isShiftPressed = false
	end
end)
